# 3. PDF 读取函数
def read_pdf(file_path="paper.pdf"):
    """
    读取本地 PDF 文件，提取所有页面的文字内容
    
    参数:
        file_path (str): PDF 文件的路径，默认为当前目录下的 "paper.pdf"
    
    返回:
        str: 提取出的全部文本内容，如果失败则返回空字符串
    
    功能说明:
        - 使用 pypdf 库逐页读取 PDF
        - 自动过滤空页面
        - 在页面间添加分隔符，便于后续处理
        - 完善的错误处理机制
    """
    print(f"📄 正在读取 PDF 文件：{file_path}")
    print("-" * 60)
    
    try:
        # 3.1 检查文件是否存在
        if not os.path.exists(file_path):
            raise FileNotFoundError(f"找不到文件：{file_path}")
        
        # 3.2 创建 PdfReader 对象
        # PdfReader 是 pypdf 库的核心类，用于读取和解析 PDF 文件
        reader = PdfReader(file_path)
        
        # 3.3 获取 PDF 的总页数
        # len(reader.pages) 返回 PDF 文档的页面总数
        total_pages = len(reader.pages)
        print(f"📊 PDF 总页数：{total_pages} 页")
        
        # 3.4 初始化一个列表，用于存储所有页面的文本
        all_text = []
        
        # 3.5 逐页提取文本内容
        for page_num in range(total_pages):
            # 获取当前页面对象
            page = reader.pages[page_num]
            
            # 从页面中提取文本
            # extract_text() 方法会自动识别并提取页面中的所有文字
            page_text = page.extract_text()
            
            # 3.6 过滤空页面并添加页码标记
            if page_text and page_text.strip():
                # 为每一页添加页码标记，方便后续引用
                marked_text = f"[第 {page_num + 1} 页]\n{page_text.strip()}"
                all_text.append(marked_text)
                
                # 显示提取进度
                text_preview = page_text[:50].replace('\n', ' ')
                print(f"✅ 第 {page_num + 1}/{total_pages} 页已提取 | 预览：{text_preview}...")
            else:
                print(f"⚠️  第 {page_num + 1}/{total_pages} 页为空或无法提取文本")
        
        # 3.7 将所有页面的文本拼接成一个完整的字符串
        # 使用双换行符分隔不同页面，保持结构清晰
        full_text = "\n\n".join(all_text)
        
        # 3.8 统计提取结果
        total_chars = len(full_text)
        print(f"\n📈 提取完成！")
        print(f"   - 有效页数：{len(all_text)}/{total_pages}")
        print(f"   - 总字符数：{total_chars}")
        print(f"   - 预估 token 数：约 {total_chars // 4} tokens")  # 粗略估算：中文约 1 token/字，英文约 1 token/4 字符
        print("-" * 60)
        print()
        
        return full_text
        
    except FileNotFoundError as e:
        print(f"❌ 文件错误：{e}")
        print("请确保 paper.pdf 文件在当前目录下")
        return ""
    except Exception as e:
        print(f"❌ PDF 读取失败：{e}")
        print("可能的原因：")
        print("  1. PDF 文件已损坏")
        print("  2. PDF 是扫描版图片（无法直接提取文本）")
        print("  3. PDF 文件受密码保护")
        return ""

    # 5.5 检查文本长度

    estimated_tokens = len(pdf_text) // 4  # 粗略估算

    print("=" * 60)

    try:
        # 5.6.1 发起流式聊天请求

            # 使用 glm-4-flash 模型（高性价比快速模型）

            # 构建消息列表

            # 启用流式输出，实现打字机效果

                # 获取文本增量

                # 实时打印到终端
